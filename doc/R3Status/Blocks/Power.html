def get_battery_path
  base_paths = ["/proc/acpi/battery/", "/sys/class/power_supply/"]
  base_paths.each do |p|
    if Dir.exist? p
      return p + (Dir.entries(p) - %w{ . .. }).last
    end
  end
  
end
