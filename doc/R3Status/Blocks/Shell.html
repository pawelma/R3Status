def clicked(button, x, y)
  super(button, x, y)
  
  case click_commands.class
  when String
    %x#{click_commands}`
  when Array
    %x#{click_commands[button - 1]}`
  when Hash
    %x#{click_commands[button]}`
  end
end

def update
  return if command.nil?
  
  cmd = %x#{command}`.split.map {|str| str.chomp}
  return if cmd.length == 0
  
  @full_text = cmd[0]
  @text_color = cmd[1] if cmd.length >= 2
end
